def sort_per_day(csv_file, day):
    lines_for_day = []
    for i in range(len(csv_file)):
        line = csv_file.iloc[i]
        #print(line["Datum"][:2])
        if line["Datum"][:2] == str(day):
            #print(line["Datum"]) 
            lines_for_day.append(line)
        elif line["Datum"][:2] == "0" + str(day):
            #print(line["Datum"])
            lines_for_day.append(line)
    #print(f"Liste mit Linien für Tag {day}: {lines_for_day}")
    return lines_for_day

# die geräteliste aber formatiert sodass man auf einzelne elemente in der gerätezeilen zugreifen kann (listein einer liste!)
def format_kolonne(log_kolonne_unformatiert):
    formatted_list = []
    for gerät in log_kolonne_unformatiert:
        # Alle Klammern entfernen
        gerät = gerät.replace("(", "").replace(")", "").replace(";", "")
        # Immer da, wo es einen Abstand hat, wird die Liste gesplittet
        gerät = gerät.split()
        formatted_list.append(gerät)
    
    return formatted_list

def unique_devices(geräte_kolonne_formatiert):
    devices = []
    for i in geräte_kolonne_formatiert:
        if len(i) > 1:
            gerätename = i[1]
            #print(gerätename)

            if not gerätename in devices:
